#### C++代码

```c++
#include <iostream>
#include <queue>

using namespace std;

struct Node
{
    int data;
    Node* left;
    Node* right;
    Node(int data)
    {
        this->data = data;
        this->left = NULL;
        this->right = NULL;
    }
};

class BinTree
{
public:
    Node* root;
    Node* CreateTree();
    void BreathFirst(Node *r);  //层序遍历
};

Node* BinTree::CreateTree()
{
    Node *p1 = new Node(1);  
    Node *p2 = new Node(2);  
    Node *p3 = new Node(3);  
    Node *p4 = new Node(4);  
    Node *p5 = new Node(5);  
    Node *p6 = new Node(6);  
    Node *p7 = new Node(7);  
    Node *p8 = new Node(8);  
    Node *p9 = new Node(9);  
    Node *p10 = new Node(10);
    Node *p11 = new Node(11);
    Node *p12 = new Node(12);
    p1->left = p2;
    p1->right = p3;
    p2->left = p4;
    p2->right = p5;
    p3->left = p6;
    p3->right = p7;
    p4->left = p8;
    p4->right = p9;
    p5->right = p10;
    p7->left = p11;
    p7->right = p12;
    root = p1;
    return p1;
}

//层序遍历(利用队列)
void BinTree::BreathFirst(Node *r)
{
    if(r == NULL)
    {
        return;
    }
    queue<Node *> q;
    q.push(r);  //根节点进队列

    while(!q.empty())   //队列不为空判断
    {
        cout << q.front()->data << " ";
        if(q.front()->left != NULL)    //如果有左孩子，左孩子加入队列
        {
            q.push(q.front()->left);
        }
        if(q.front()->right != NULL)  //如果有右孩子，右孩子加入队列
        {
            q.push(q.front()->right);
        }
        q.pop();
    }
    cout << endl;
}

int main()
{
    BinTree t;
    t.CreateTree()；
    cout << "层序遍历：";
    t.BreathFirst(t.root);
}
```
